\chapter{Keyboard Shortcuts}

\section{Zoom Keyboard Shortcuts}

\begin{tabular}{| c | c |}
\hline
Shortcut & Description \\
\hline
CTRL-S & This will stop sharing the Zoom screen (zoom must be selected) \\
ALT-SHIFT-H & This will remove the Zoom bar on the top of the shared screen. \\
\hline
\end{tabular}
