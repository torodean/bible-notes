% Chapter 23 of Genesis
\bookchapter{\name{Sarah}'s Death and Burial}

\bverse \name{Sarah} lived one hundred and twenty-seven years; \textit{these were} the years of the life of \name{Sarah}.
\bverse So \name{Sarah} died in \location{Kirjath Arba} (that \textit{is}, \location{Hebron}) in the land of \location{Canaan}, and \name{Abraham} came to mourn for \name{Sarah} and to weep for her.
\bverse Then \name{Abraham} stood up from before his dead, and spoke to the sons of \name{Heth}, saying
\bverse ``I \textit{am} a foreigner and a visitor among you. Give me property or a burial place among you, that I may bury my dead out of my sight.''
\bverse And the sons of \name{Heth} answered \name{Abraham}, saying to him, 
\bverse ``Hear us, my lord: You \are a mighty prince among us; bury your dead in the choicest of our burial places. None of us will withhold from you his burial place, that you may bury your dead.''
\bverse Then \name{Abraham} stood up and bowed himself to the people of the land, the sons of \name{Heth}.
\bverse And he spoke with them, saying, ``If it is your wish that I bury my dead out of my sight, hear me, and meet with \name{Ephron} the son of \name{Zohar} for me,
\bverse that he may give me the cave of \name{Machpelah} which he has, which \textit{is} at the end of his field. Let him give it to me at the full price, as property for a burial place among you.''
\bverse Now \name{Ephron} dwelt among the sons of \name{Heth}; and \name{Ephron} the \people{Hittite} answered \name{Abraham} in the presence of the sons of \name{Heth}, all who entered at the gate of his city, saying,
\bverse ``No, my lord, hear me: I give you the field and the cave that \textit{is} in it; I give it to you in the presence of the sons of my people. I give it to you. Bury your dead!''
\bverse Then \name{Abraham} bowed himself down before the people of the land;
\bverse and he spoke to \name{Ephron} in the hearing of the people of the land, saying, ``If you \text{will give it}, please hear me. I will give you money for the field; take \textit{it} from me and I will bury me dead there.''
\bverse And \name{Ephron} answered \name{Abraham}, saying to him,
\bverse ``My lord, listen to me; the land \textit{is worth} four hundred shekels of silver. What \textit{is} that between you and me? So bury your dead.''
\bverse And \name{Abraham} listened to \name{Ephron}; and \name{Abraham} weighed out the silver for \name{Ephron} which he had named in the hearing of the sons of \name{Heth}, four hundred shekels of silver, currency of the merchants.
