% Chapter 10 of Genesis
\bookchapter{Nations Descended from \name{Noah}}

\bverse Now this \textit{is} the genealogy of the sons of \name{Noah}: \name{Shem}, \name{Ham}, and \name{Japheth}. And sons were born to them after the flood.
\bverse The sons of \name{Japheth} \were \name{Gomer}, \name{Magog}, \name{Madai}, \name{Javan}, \name{Tubal}, \name{Meshech}, and \name{Tiras}.
\bverse The sons of \name{Gomer} \were \name{Ashkenaz}, \name{Riphath}, and \name{Togarmah}.
\bverse The sons of \name{Javan} \were \name{Elishah}, \name{Tarshish}, \name{Kittim}, and \name{Dodanim}.
\bverse From these the coastland \textit{peoples} of the Gentiles were separated into their lands, everyone according to his tongue\vmark{a}, according to their families, into their nations.
	\translationnote{a}{The word for `tongue' here (Strong's 3956 \cite{Strong's GodRules}) is often translated as language, but is distinct from the term for language used in \bref{Genesis 11:1} (thus translated to tongue here). this means tongue (literal or \textit{organ of speech}), language, etc. It could refer to various dialects, or different languages entirely.}

\bverse The sons of \name{Ham} \were \name{Cush}, \name{Mizraim}, \name{Put}, and \name{Canaan}.
\bverse The sons of \name{Cush} \were \name{Seba}, \name{Havilah}, \name{Sabtah}, \name{Raamah}, and \name{Sabtechah}; and the sons of \name{Raamah} \were \name{Sheba} and \name{Dedan}.
\bverse \name{Cush} begot \name{Nimrod}; he began to be a mighty one on the earth.
\bverse He was a mighty hunter before the \lord; therefore it is said, ``Like \name{Nimrod} the mighty hunter before the \lord.
\bverse And the beginning of his kingdom was Babel, Erech, Accad, and Calneh, in the land of Shinar.
\bverse From that land he went to Assyria and built Nineveh, Rehoboth Ir, Calah,
\bverse and Resen between Nineveh and Calah (that \textit{is} the principal city).
\bverse \name{Mizraim} begot \name{Ludim}, \name{Anamim}, \name{Lehabim}, \name{Naphtuhim},
\bverse \name{Pathrusim}, and \name{Casluhim} (from whom came the Philistines and Caphtorim).
\bverse \name{Canaan} begot \name{Sidon} his firstborn, and \name{Heth};
\bverse the Jebusite, the Amorite, and the Girgashite;
\bverse The Hivite, the Arkite, and the Sinite;
\bverse The Arvadite, the Zemarite, and the Hamathite. Afterward the families of the Canaanites were dispersed.
\bverse And the border of the Canaanites was from Sidon as you go toward Gerar, as far as Gaza; then as you go toward Sodom, Gomorrah, Admah, and Zeboiim, as far as Lasha.
\bverse These \were the sons of \name{Ham}, according to their families, according to their tongues, in their lands \textit{and} in their nations.
\bverse And \textit{children} were born also to \name{Shem}, the father of all the children of \name{Eber}, the brother of \name{Japheth} the elder\vmark{a}.
	\translationnote{a}{The word `elder` here (Strong's 1419 \cite{Strong's GodRules}) could be referring to the older of age or of in importance (distinguished) such as a church elder.}

\bverse The sons of \name{Shem} \were \name{Elam}, \name{Asshur}, \name{Arphaxad}, \name{Lud}, and \name{Aram}.
\bverse The sons of \name{Aram} \were \name{Uz}, \name{Hul}, \name{Gether}, and \name{Mash}.
\bverse \name{Arphaxad} begot \name{Salah}, and \name{Salah} begot \name{Eber}.
\bverse To \name{Eber} were born two sons: the name of one \was \name{Peleg}, for in his days the earth was divided; and his brother's name \was \name{Joktan}.
\bverse \name{Joktan} begot \name{Almodad}, \name{Sheleph}, \name{Hazarmaveth}, \name{Jerah},
\bverse \name{Hadoram}, \name{Uzal}, \name{Diklah},
\bverse \name{Obal}, \name{Abimael}, \name{Sheba},
\bverse \name{Ophir}, \name{Havilah}, and \name{Jobab}. All these \were the sons of \name{Joktan}.
\bverse And their dwelling place was from Mesha as you go toward Sephar, the mountain of the east.
\bverse These \were the sons of \name{Shem}, according to their families, according to their tongues, in their lands, according to their nations.
\bverse These \were the families of the sons of \name{Noah}, according to their generations, in their nations; and from thee the nations were divided on the earth after the flood.
