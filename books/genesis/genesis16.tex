% Chapter 16 of Genesis
\bookchapter{\name{Hagar} and \name{Ishmael}}

\bverse Now \name{Sarai}, \name{Abram}'s wife, had borne him no \textit{children}. And she had an \people{Egyptian} maidservant whose name was \name{Hagar}.
\bverse So \name{Sarai} said to \name{Abram}, ``See now, the \lord has restrained me from bearing \textit{children}. Please, go in to my maid; perhaps I shall obtain children by her.'' And \name{Abram} heeded the voice of \name{Sarai}.
\bverse Then \name{Sarai}, \name{Abram}'s wife, took \name{Hagar} her maid, the \people{Egyptian}, and gave her to her husband \name{Abram} to be his wife, after \name{Abram} had dwelt ten years in the land of \location{Canaan}.
\bverse So he went in to \name{Hagar}, and she conceived. And when she saw that she had conceived, her mistress became despised in her eyes.
	\generalnote{}{This is a good example of Jealousy. It demonstrates one reason that polygamy is dangerous. Even though \name{Sarai} gave \name{Abram} permission and pleaded with him to conceive with \name{Hagar}, contentions grew between them because of the situation.}

\bverse Then \name{Sarai} said to \name{Abram}, ``My wrong \textit{be} upon you! I gave my maid into your embrace; and when she saw that she had conceived, I became despised in her eyes. The \lord judge between you and me.''
\bverse So \name{Abram} said to \name{Sarai}, ``Indeed your maid \textit{is} in your hand; do to her as you please.'' And when \name{Sarai} dealt harshly with her, she fled from her presence.
\bverse Now the Angel of the \lord found her by a spring of water in the wilderness, by the spring on the way to \location{Shur}.
\bverse And He said, ``\name{Hagar}, \name{Sarai}'s maid, where have you come from, and where are you going?'' She said, ``I am fleeing from the presence of my mistress \name{Sarai}.''
\bverse The Angel of the \lord said to her, ``Return to your mistress, and submit yourself under her hand.''
\bverse Then the Angel of the \lord said to her, ``I will multiply your descendants exceedingly, so that they shall not be counted for multitude.''
\bverse And the Angel of the \lord said to her:
\begin{bquotation}
``Behold, you \are with child, And you shall bear a son. You shall call his name Ishmael, Because the \lord has heard your affliction. \bverse He shall be a wild man; His hand \textit{shall be} against every man, And every man's hand against him. And he shall dwell in the presence of all his brethren.''
\end{bquotation}

\bverse Then she called the name of the \lord who spoke to her, You-Are-the-God-Who-Sees; for she said, ``Have I also here seen Him who sees me?''
\bverse Therefore the well was called Beer Lahai Roi; observe, \textit{it is} between \location{Kadesh} and \location{Bered}.
\bverse So \name{Hagar} bore \name{Abram} a son; and \name{Abram} named his son, whom \name{Hagar} bore, \name{Ishmael}.
\bverse \name{Abram} \was eighty-six years old when \name{Hagar} bore \name{Ishmael} to \name{Abram}.
