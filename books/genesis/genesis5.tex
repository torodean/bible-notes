% Chapter 5 of Genesis
\bookchapter{The Family Of \name{Adam}}


\bverse This is the book of the genealogy\vmark{a} of \name{Adam}. In the day that God created man, He made him in the likeness of God.
	\contextnote{a}{Only the names of those leading to the genealogy of \name{Noah} are given. The others are not mentioned, but it is implied they exist in \bref{Genesis 5:4}, \bref{Genesis 5:7}, \bref{Genesis 5:10} and other scriptures.}

\bverse He created them male and female, and blessed them and called them Mankind in the day they were created.
\bverse And \name{Adam}\vmark{a} lived one hundred and thirty years, and begot \textit{a son} in his own likeness, after his image, and named him \name{Seth}.
	\translationnote{a}{The word/name `\name{Adam}' (Strong's 121\cite{Strong's GodRules}) means `red.' Other sources say this means `from red soil', `son of the red Earth\index{Earth}', or similar translations.}

\bverse After he begot \name{Seth}, the days of \name{Adam} were eight hundred years; and he had sons and daughters.
\bverse So all the days that \name{Adam} lived were nine hundred and thirty years; and he died.
\bverse \name{Seth}\vmark{a} lived one hundred and five years, and begot \name{Enosh}.
	\translationnote{a}{The word/name `\name{Seth}' (Strong's 8352\cite{Strong's GodRules}) means `compensation.'}
	
\bverse After he begot \name{Enosh}, \name{Seth} lived eight hundred and seven years, and had sons and daughters.	
\bverse So all the days of \name{Seth} were nine hundred and twelve years; and he died.
\bverse \name{Enosh}\vmark{a} lived ninety years, and begot \name{Cainan}.
	\translationnote{a}{The word/name `\name{Enosh}' (Strong's 583\cite{Strong's GodRules}) means `man.'}
	
\bverse After he begot \name{Cainan}, \name{Enosh} lived eight hundred and fifteen years, and had sons and daughters.	
\bverse So all the days of \name{Enosh} were nine hundred and five years; and he died.
\bverse \name{Cainan}\vmark{a} lived seventy years, and begot \name{Mahalalel}.
	\translationnote{a}{The word/name `\name{Cainan}' (Strong's 7018\cite{Strong's GodRules}) means `possession.'}
	
\bverse After he begot \name{Mahalalel}, \name{Cainan} lived eight hundred and forty years, and had sons and daughters.	
\bverse So all the days of \name{Cainan} were nine hundred and ten years; and he died.
\bverse \name{Mahalalel}\vmark{a} lived sixty-five years, and begot \name{Jared}. 
	\translationnote{a}{The word/name `\name{Mahalalel}' (Strong's 4111\cite{Strong's GodRules}) means `praise of God.'}
	
\bverse After he begot \name{Jared}, \name{Mahalalel} lived eight hundred and thirty years, and had sons and daughters. 
\bverse So all the days of \name{Mahalalel} were eight hundred and ninety-five years; and he died.
\bverse \name{Jared}\vmark{a} lived one hundred and sixty-two years, and begot \name{Enoch}.
	\translationnote{a}{The word/name `\name{Jared}' (Strong's 3382\cite{Strong's GodRules}) means `descent.'}
	
\bverse After he begot \name{Enoch}, \name{Jared} lived eight hundred years and had sons and daughters.
\bverse So all the days of \name{Jared} were nine hundred and sixty-two years; and he died.
\bverse \name{Enoch}\vmark{a} lived sixty-five years, and begot \name{Methuselah}.
	\translationnote{a}{The word/name `\name{Enoch}' (Strong's 2585\cite{Strong's GodRules}) means `dedicated.'}
	
\bverse After he begot \name{Methuselah}, \name{Enoch} walked with God three hundred years, and had sons and daughters.
\bverse Soo all the days of \name{Enoch} were three hundred and sixty-five years.
\bverse And \name{Enoch} walked with God; and he \was not, for God took him.
	\questionnote{}{This scripture breaks the pattern seen with all the other members of this genealogy. With the others, their total days are numbered and they eventually die. However, with \name{Enoch}, it simply says `God took him' \textit{instead} of `he died.' What does this mean? Does this mean that God took him to another place, or that God took him in death early? Perhaps this can be a distinction between dying of old age (like the others) and dying of unnatural causes (in this case). This somehow has to fit with \bref{Hebrews 9:27} and \bref{John 3:13}.}

\bverse \name{Methuselah}\vmark{a} lived one hundred and eighty-seven years, and begot \name{Lamech}.
	\translationnote{a}{The word/name `\name{Methuselah}' (Strong's 4968\cite{Strong's GodRules}) means `man of the dart.'}
	
\bverse After he begot \name{Lamech}, \name{Methuselah} lived seven hundred and eighty-two years, and had sons and daughters.
\bverse So all the days of \name{Methuselah} were nine hundred and sixty-nine years; and he died.
\bverse \name{Lamech}\vmark{a} lived one hundred and eighty-two years, and had a son.
	\translationnote{a}{The word/name `\name{Lamech}' (Strong's 3929\cite{Strong's GodRules}) means `powerful.'}
	
\bverse And he called his name \name{Noah}\vmark{a}, saying, ``This \textit{one} will comfort us concerning our work and the toil of our hands, because of the ground which the \lord has cursed.''
	\translationnote{a}{The word/name `\name{Noah}' (Strong's 5146\cite{Strong's GodRules}) means `rest.'}
	
\bverse After he begot \name{Noah}, \name{Lamech} lived five hundred and ninety-five years, and had sons and daughters.
\bverse So all the days of \name{Lamech} were seven hundred and seventy-seven years; and he died.
\bverse And \name{Noah} was five hundred years old, and \name{Noah} begot \name{Shem}\vmark{a}, \name{Ham}\vmark{b}, and \name{Japheth}\vmark{c}.
	\translationnote{a}{The word/name `\name{Shem}' (Strong's 8035\cite{Strong's GodRules}) means `name.'}
	\translationnote{a}{The word/name `\name{Ham}' (Strong's 2526\cite{Strong's GodRules}) means `hot.'}
	\translationnote{a}{The word/name `\name{Japheth}' (Strong's 3315\cite{Strong's GodRules}) means `opened.'}
